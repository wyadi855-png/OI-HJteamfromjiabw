P3376 最大流模板题详解
张伟 2024-11-19 P3376 → luogu 网络流

题目思路

本题是最大流的模板题，使用 Dinic 算法。核心分两步：①用 BFS 构造分层图；②在分层图上用 DFS 多路增广。每次 BFS + DFS 为一轮，直到源点无法到达汇点为止。

C++ 代码

#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9;
struct Edge { int to, rev; long long cap; };
vector<Edge> graph[205];
int level[205], iter[205];

void add_edge(int from, int to, long long cap) {
    graph[from].push_back({to, (int)graph[to].size(), cap});
    graph[to].push_back({from, (int)graph[from].size()-1, 0});
}

bool bfs(int s, int t, int n) {
    fill(level, level+n+1, -1);
    queue<int> q;
    level[s] = 0; q.push(s);
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (auto& e : graph[v])
            if (e.cap > 0 && level[e.to] < 0)
                level[e.to] = level[v]+1, q.push(e.to);
    }
    return level[t] >= 0;
}

long long dfs(int v, int t, long long f) {
    if (v == t) return f;
    for (int& i = iter[v]; i < (int)graph[v].size(); i++) {
        Edge& e = graph[v][i];
        if (e.cap > 0 && level[v] < level[e.to]) {
            long long d = dfs(e.to, t, min(f, e.cap));
            if (d > 0) { e.cap -= d; graph[e.to][e.rev].cap += d; return d; }
        }
    }
    return 0;
}

long long max_flow(int s, int t, int n) {
    long long flow = 0;
    while (bfs(s, t, n)) {
        fill(iter, iter+n+1, 0);
        long long d;
        while ((d = dfs(s, t, INF)) > 0) flow += d;
    }
    return flow;
}

int main() {
    int n, m, s, t;
    cin >> n >> m >> s >> t;
    for (int i = 0; i < m; i++) {
        int u, v; long long w;
        cin >> u >> v >> w;
        add_edge(u, v, w);
    }
    cout << max_flow(s, t, n) << endl;
}

反思与总结

Dinic 最关键的优化是「当前弧优化」，用 iter[] 记录下次搜索的起始边，避免重复搜索已经饱和的边。建议背诵这个模板，在比赛中直接套用，不要浪费时间重写。

树形 DP 从入门到 P1352
林思远 2024-10-25 P1352 → luogu 树形DP

题目思路

经典树形 DP。定义 dp[u][0/1] 表示以 u 为根的子树中，u 不参加/参加舞会时的最大快乐值。转移枚举所有子节点 v，若 u 不参加则 v 可以参加也可以不参加，取最大值；若 u 参加则 v 一定不参加。

C++ 代码

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 6005;
int n, w[MAXN], dp[MAXN][2];
vector<int> ch[MAXN];
bool has_parent[MAXN];

void dfs(int u) {
    dp[u][0] = 0;
    dp[u][1] = w[u];
    for (int v : ch[u]) {
        dfs(v);
        dp[u][0] += max(dp[v][0], dp[v][1]);
        dp[u][1] += dp[v][0];
    }
}

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> w[i];
    for (int i = 1; i < n; i++) {
        int l, k; cin >> l >> k;
        ch[k].push_back(l);
        has_parent[l] = true;
    }
    int root = 1;
    while (has_parent[root]) root++;
    dfs(root);
    cout << max(dp[root][0], dp[root][1]) << endl;
}

反思与总结

树形 DP 一定要先想清楚状态定义再写代码。这道题状态很简单，但稍微复杂一点的题（如树背包）就需要更小心地考虑复杂度。

Tarjan 求强连通分量
陈晓宇 2024-10-18 P3387 → luogu 图论

题目思路

P3387 是缩点模板题：先用 Tarjan 求所有 SCC，将每个 SCC 缩成一个点，建新的 DAG，然后在 DAG 上跑 DP 求最大权路径。

C++ 代码

#include <bits/stdc++.h>
using namespace std;

int n, m, dfn[1e4+5], low[1e4+5], scc_id[1e4+5];
int timer_=0, scc_cnt=0;
long long scc_val[1e4+5], dp[1e4+5];
bool in_stack[1e4+5];
stack<int> st;
vector<int> G[1e4+5], dag[1e4+5];
int val[1e4+5], in_deg[1e4+5];

void tarjan(int u) {
    dfn[u] = low[u] = ++timer_;
    st.push(u); in_stack[u] = true;
    for (int v : G[u]) {
        if (!dfn[v]) { tarjan(v); low[u] = min(low[u], low[v]); }
        else if (in_stack[v]) low[u] = min(low[u], dfn[v]);
    }
    if (low[u] == dfn[u]) {
        scc_cnt++;
        while (true) {
            int v = st.top(); st.pop();
            in_stack[v] = false;
            scc_id[v] = scc_cnt;
            scc_val[scc_cnt] += val[v];
            if (v == u) break;
        }
    }
}
// ... (DAG DP 省略)

反思与总结

Tarjan 的细节很多，尤其是 low 值的更新：只有在栈中的节点才能用 dfn[v] 更新，已经出栈的（属于其他 SCC 的）不能更新。这一点一定要牢记。
